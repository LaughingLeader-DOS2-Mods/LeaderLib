Version 1
SubGoalCombiner SGC_AND
INITSECTION
//The version is registered in LeaderLib_00_1_1_System, so other mods will be able to see if LeaderLib is registered.
KBSECTION
//REGION UPDATING
IF
StoryEvent(_,"LeaderLib_Events_Internal_OnUpdateSettings")
THEN
LeaderLog_Log("SYSTEM", "[LeaderLib:Updater] Settings updated.");
//END_REGION

//REGION UPDATE_EVENT
IF
GameStarted(_,_)
THEN
LeaderLog_Log("SYSTEM", "[LeaderLib:Updater] Game started.");

//REGION Whoops
IF
SavegameLoading(_,_,_,_)
AND
DB_Mods_Registered("LeaderLib", "LaughingLeader", "1.5.57.0")
THEN
NOT DB_LeaderLib_StringExt_VersionToInt("1.5.57.0", 1, 5, 57, 0);
NOT DB_Mods_Registered("LeaderLib", "LaughingLeader", "1.5.57.0");
NOT DB_LeaderUpdater_Temp_PastRegistered("LeaderLib", "LaughingLeader", "1.5.57.0");
DB_LeaderLib_StringExt_VersionToInt("1.5.7.0", 1, 5, 7, 0);
DB_LeaderLib_StringExt_VersionToInt("1.5.7.1", 1, 5, 7, 1);
DB_Mods_Registered("LeaderLib", "LaughingLeader", "1.5.7.0");
DB_LeaderUpdater_Temp_PastRegistered("LeaderLib", "LaughingLeader", "1.5.7.0");
//END_REGION

IF
GameStarted(_,_)
AND
LeaderUpdater_QRY_ModUpdateNeeded("LeaderLib", "LaughingLeader", 1,5,9,3)
THEN
LeaderLog_Log("SYSTEM", "[LeaderLib:Updater] Updating LeaderLib to latest version (1.5.9.3).");
LeaderUpdater_Register_Mod("LeaderLib", "LaughingLeader", 1,5,9,3);
LeaderLib_SendEvent("LeaderLib_Events_Internal_OnUpdateSettings");
//LeaderLib_StringExt_Internal_ClearCache_Lazy(100);
//END_REGION

//REGION ACTIVE_FLAG
IF
GameStarted(_,_)
AND
GlobalGetFlag("LeaderLib_Active", 0)
THEN
GlobalSetFlag("LeaderLib_Active");
//END_REGION

//REGION GENERAL_UPDATES
PROC
LeaderUpdater_ModUpdated("LeaderLib", "LaughingLeader", (STRING)_PastVersion, (STRING)_NewVersion)
AND
LeaderLib_StringExt_QRY_VersionIsLessThan(_PastVersion, 1,5,9,1)
THEN
LeaderUpdater_Register_UUID("LeaderLib", "LaughingLeader", 7e737d2f-31d2-4751-963f-be6ccc59cd0c);
//END_REGION
EXITSECTION

ENDEXITSECTION
ParentTargetEdge "LaughingLeader__LeaderLib"
