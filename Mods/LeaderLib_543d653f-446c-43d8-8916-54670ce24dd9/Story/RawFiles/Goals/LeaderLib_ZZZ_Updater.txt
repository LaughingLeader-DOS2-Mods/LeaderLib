Version 1
SubGoalCombiner SGC_AND
INITSECTION
//The version is registered in LeaderLib_00_1_1_System, so other mods will be able to see if LeaderLib is registered.
KBSECTION
//REGION UPDATE_EVENT
// [BEGIN_NO_OSITOOLS]
IF
GameStarted(_,_)
AND
LeaderUpdater_QRY_ModUpdateNeeded("LeaderLib", "LaughingLeader", 1,6,0,0)
THEN
LeaderLog_Log("SYSTEM", "[LeaderLib:Updater] Updating LeaderLib to latest version (1.6.0.0).");
LeaderUpdater_Register_Mod("LeaderLib", "LaughingLeader", 1,6,0,0);
// [END_NO_OSITOOLS]
//END_REGION

//REGION ACTIVE_FLAG
IF
GameStarted(_,_)
AND
GlobalGetFlag("LeaderLib_Active", 0)
THEN
GlobalSetFlag("LeaderLib_Active");
//END_REGION

//REGION GENERAL_UPDATES
PROC
LeaderUpdater_ModUpdated("LeaderLib", "LaughingLeader", (STRING)_PastVersion, (STRING)_NewVersion)
AND
LeaderLib_StringExt_QRY_VersionIsLessThan(_PastVersion, 1,5,9,1)
THEN
LeaderUpdater_Register_UUID("LeaderLib", "LaughingLeader", 7e737d2f-31d2-4751-963f-be6ccc59cd0c);
//END_REGION
EXITSECTION

ENDEXITSECTION
ParentTargetEdge "LaughingLeader__LeaderLib"
